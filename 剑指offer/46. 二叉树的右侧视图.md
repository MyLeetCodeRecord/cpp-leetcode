#### 46. [二叉树的右侧视图](https://leetcode.cn/problems/WNC0Lk/):  `(每次遍历整层的)BFS`

```CPP
vector<int> rightSideView(TreeNode* root) {
    if(root==NULL)
        return {};
    vector<int> ans;
    queue<TreeNode*> q;
    q.push(root);
    while(!q.empty()){
        int _size = q.size();
        for(int i=0; i<_size; i++){
            TreeNode* cur = q.front();
            if(i == _size-1)
                ans.push_back(cur->val);
            q.pop();
            if(cur->left!=NULL)
                q.push(cur->left);
            if(cur->right!=NULL)
                q.push(cur->right);
        }
    }
    return ans;
}
```
