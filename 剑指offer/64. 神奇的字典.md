#### 64. [神奇的字典](https://leetcode.cn/problems/US1pGT/?favorite=e8X3pBZi): `Trie` `DFS`

> "必须修改一个字母" => `DFS(changed, ...)` & `DFS(changed+1, ...)`

```CPP
struct Trie{
    Trie* child[26];
    bool isEnd;
    Trie(){
        for(int i=0; i<26; i++)
            child[i] = NULL;
        isEnd = false;
    }
};
class MagicDictionary {
private:
    Trie* trie;
public:
    MagicDictionary() {
    }
    void add(string word){
        Trie* t = trie;
        for(char ch: word){
            if(t->child[ch-'a']==NULL)
                t->child[ch-'a'] = new Trie();
            t = t->child[ch-'a'];
        }
        t->isEnd = true;
    }
    // 类似“带通配符的search” - DFS
    bool dfs_search(string word, int changed, int cur, Trie* t){
        if(changed > 1)
            return false;
        if(cur == word.size())
            return t->isEnd && changed==1;
        for(int i=0; i<26; i++){
            if(t->child[i] != NULL){
                // 不用修改
                if(i==word[cur]-'a'){
                    if(dfs_search(word, changed, cur+1, t->child[i]))
                        return true;
                }
                // 需要修改
                else if(changed==0 && dfs_search(word, changed+1, cur+1, t->child[i]))
                    return true;
            }
        }
        return false;
    }
    bool search(string searchWord) {
        return dfs_search(searchWord, 0, 0, trie);
    }
    void buildDict(vector<string> dictionary) {
        trie = new Trie();
        for(string word: dictionary){
            add(word);
        }
    }
};
```
