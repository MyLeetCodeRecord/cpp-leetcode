#### 62.[实现前缀树](https://leetcode.cn/problems/QC3q1f/?favorite=e8X3pBZi): `Trie`

```CPP
struct TrieNode{
    bool isEnd;
    TrieNode* child[26];
    TrieNode(){
        isEnd = false;
        for(int i=0; i<26; i++)
            child[i] = NULL;
    }
};
class Trie {
private:
    TrieNode* trie;
public:
    Trie() {
        trie = new TrieNode();
    }
    void insert(string word) {
        TrieNode* t = this->trie;
        for(int i=0; i<word.size(); i++){
            if(t->child[word[i]-'a']==NULL)
                t->child[word[i]-'a'] = new TrieNode();
            t = t->child[word[i]-'a'];
        }
        t->isEnd = true;
    }
    bool search(string word) {
        TrieNode* t = this->trie;
        for(int i=0; i<word.size(); i++){
            if(t->child[word[i]-'a']==NULL)
                return false;
            t = t->child[word[i]-'a'];
        }
        return t->isEnd;
    }
    bool startsWith(string prefix) {
        TrieNode* t = this->trie;
        for(int i=0; i<prefix.size(); i++){
            if(t->child[prefix[i]-'a']==NULL)
                return false;
            t = t->child[prefix[i]-'a'];
        }
        return true;
    }
};
```
