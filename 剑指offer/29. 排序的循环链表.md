#### 29. [排序的循环链表](https://leetcode.cn/problems/4ueAj6/): `环形链表`

> ![LC708](/appendix/LC708.jpg)

```CPP
void insertNode(Node* pre, int val){
    Node* inserted = new Node(val);
    inserted->next = pre->next;
    pre->next = inserted;
}
Node* insert(Node* head, int insertVal) {
    // 单独判断空链表, 插入后记得连成环
    if(head==NULL){
        head = new Node(insertVal);
        head->next = head;
        return head;
    }
    Node* cur = head;
    bool flag = false;
    // 为什么是cur->next != head ?
    // 相当于一共n个区间, 检查前n-1个都不是, 就放到n-1里面
    while(cur->next != head){
        // 特殊区间 4->1
        if(cur->val > cur->next->val){
            if(insertVal >= cur->val || insertVal <= cur->next->val){
                insertNode(cur, insertVal);
                flag = true;
                break;
            }
            else
                cur = cur->next;
        }
        // 普通区间 1->3
        else{
            if(insertVal >= cur->val && insertVal <=cur->next->val){
                insertNode(cur, insertVal);
                flag = true;
                break;
            }
            else
                cur = cur->next;
        }
    }
    if(cur->next==head && !flag){
        Node* inserted = new Node(insertVal);
        inserted->next = cur->next;
        cur->next = inserted;
    }
    return head;
}
```
