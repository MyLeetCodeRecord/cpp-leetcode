### 递推

#### [Acwing3777. 砖块](https://www.acwing.com/problem/content/3780/): `一次遍历` `递推`

```CPP
#include <cstdio>
#include <string>
#include <vector>
using namespace std;

bool reverseTile(string s, char target){
    vector<int> ans;
    for(int i=0; i<s.size()-1; i++){
        if(s[i] != target){
            ans.push_back(i+1);
            s[i+1] = (s[i+1]=='W')?'B':'W';
        }
    }
    if(s[s.size()-1]==target){
        printf("%d\n", (int)ans.size());
        if(ans.size()!=0){
            for(int x: ans)
                printf("%d ", x);
            printf("\n");
        }
    }
    return s[s.size()-1]==target;   // 最后一个字符不是target则说明无解
}
int main(){
    int n, k;
    scanf("%d", &n);
    for(int i=0; i<n; i++){
        scanf("%d", &k);
        string s;
        cin>>s;
        // 没有要求最短, 所以这里短路就可以了
        if(!reverseTile(s, 'B') && !reverseTile(s, 'W')){
            printf("-1\n");
        }
    }
    return 0;
}
```
